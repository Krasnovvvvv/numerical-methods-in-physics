**–†–∏—Å—É–Ω–æ–∫ 5 - –ö –ø–æ—Å—Ç–∞–Ω–æ–≤–∫–µ –∑–∞–¥–∞—á–∏ –æ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–∏ —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—ã**

#### –ü–æ—Å—Ç–∞–Ω–æ–≤–∫–∞ –∫—Ä–∞–µ–≤–æ–π –∑–∞–¥–∞—á–∏ üéØ

–ü–æ–ª—É—á–∏–º —É—Ä–∞–≤–Ω–µ–Ω–∏–µ, –æ–ø–∏—Å—ã–≤–∞—é—â–µ–µ –ø—Ä–æ—Ü–µ—Å—Å —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—ã –≤ –¥–∞–Ω–Ω–æ–π —Å–∏—Å—Ç–µ–º–µ.

–†–∞—Å—Å–º–æ—Ç—Ä–∏–º —É—á–∞—Å—Ç–æ–∫ –º–µ–∂–¥—É $x$ –∏ $x + dx$. –¢–æ–≥–¥–∞ –≤—Ö–æ–¥—è—â–∏–π –∏ –≤—ã—Ö–æ–¥—è—â–∏–π –ø–æ—Ç–æ–∫ –∏–º–µ—é—Ç –≤–∏–¥:

$$
    q_x = -kA \frac{dT}{dx} - –≤—Ö–æ–¥—è—â–∏–π
$$

$$
    q_{x+dx} = -kA\left. \frac{dT}{dx} \right|_{x+dx} - –≤—ã—Ö–æ–¥—è—â–∏–π 
$$

–≥–¥–µ $k$ - –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç —Ç–µ–ø–ª–æ–ø—Ä–æ–≤–æ–¥–Ω–æ—Å—Ç–∏, $A$ - –ø–ª–æ—â–∞–¥—å –ø–æ–ø–µ—Ä–µ—á–Ω–æ–≥–æ —Å–µ—á–µ–Ω–∏—è –ø–ª–∞—Å—Ç–∏–Ω—ã.

–ó–∞–¥–∞–¥–∏–º –ø–æ—Ç–µ—Ä–∏ –Ω–∞ –∫–æ–Ω–≤–µ–∫—Ü–∏—é:

$$
    hPdx(T(x) - T_{\infty})
$$

–≥–¥–µ $h$ - –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç —Ç–µ–ø–ª–æ–æ—Ç–¥–∞—á–∏, $–†$ - –ø–µ—Ä–∏–º–µ—Ç—Ä –ø–æ–ø–µ—Ä–µ—á–Ω–æ–≥–æ —Å–µ—á–µ–Ω–∏—è –ø–ª–∞—Å—Ç–∏–Ω—ã, $T_{\infty}$ - —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞ –æ–∫—Ä—É–∂–∞—é—â–µ–π —Å—Ä–µ–¥—ã.

–ó–∞–∫–æ–Ω —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è —ç–Ω–µ—Ä–≥–∏–∏:

$$
    q_x - q_{x+dx} - hPdx(T(x) - T_{\infty}) = 0
$$

–¢–æ–≥–¥–∞, –ø–æ–¥—Å—Ç–∞–≤–ª—è—è –≤—ã—Ä–∞–∂–µ–Ω–∏—è –¥–ª—è –ø–æ—Ç–æ–∫–æ–≤, –ø–æ–ª—É—á–∏–º:

$$
    kA[\left. \frac{dT}{dx} \right|_{x+dx} - \left. \frac{dT}{dx} \right|_{x}] = hPdx(T(x) - T_{\infty})
$$

–ü–æ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—é –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ –≤ —Ç–æ—á–∫–µ:

$$
    \left. \frac{dT}{dx} \right|_{x+dx} - \left. \frac{dT}{dx} \right|_{x} \approx \frac{d^2T}{dx^2}dx
$$

–¢–∞–∫–∏–º –æ–±—Ä–∞–∑–æ–º, –∏—Ç–æ–≥–æ–≤–æ–µ —É—Ä–∞–≤–Ω–µ–Ω–∏–µ, –æ–ø–∏—Å—ã–≤–∞—é—â–µ–µ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—ã –≤ –¥–∞–Ω–Ω–æ–π —Å–∏—Å—Ç–µ–º–µ –∏–º–µ–µ—Ç –≤–∏–¥:

$$
    \frac{d^2T}{dx^2} - \frac{hP}{kA}(T(x) - T_{\infty}) = 0,\quad x \in [0, L]
$$

–ì—Ä–∞–Ω–∏—á–Ω—ã–µ —É—Å–ª–æ–≤–∏—è:

$$
    T(0) = T_b
$$

$$
    T'(L) = 0
$$

#### –û–±–µ–∑—Ä–∞–∑–º–µ—Ä–∏–≤–∞–Ω–∏–µ –∑–∞–¥–∞—á–∏ üß©

–ü—Ä–µ–∂–¥–µ —á–µ–º —Ä–µ—à–∞—Ç—å –∫–æ–ª—á–µ–Ω–Ω—É—é –∫—Ä–∞–µ–≤—É—é –∑–∞–¥–∞—á—É, —Å —Ü–µ–ª—å—é –ø–æ–ª—É—á–µ–Ω–∏—è —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω–æ–≥–æ —Ä–µ—à–µ–Ω–∏—è, –∑–∞–≤–∏—Å—è—â–µ–≥–æ —Ç–æ–ª—å–∫–æ –æ—Ç _—á–∏—Å–ª–∞ –ë–∏–æ—Ç–∞_, –∞ –Ω–µ –æ—Ç –∫–æ–Ω–∫—Ä–µ—Ç–Ω—ã—Ö —Ñ–∏–∑–∏—á–µ—Å–∫–∏—Ö –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤ —Å–∏—Å—Ç–µ–º—ã, –ø—Ä–∏–≤–µ–¥—ë–º –µ—ë –∫ –±–µ–∑—Ä–∞–∑–º–µ—Ä–Ω–æ–º—É –≤–∏–¥—É.

_–ß–∏—Å–ª–æ –ë–∏–æ—Ç–∞_ –æ—Ç—Ä–∞–∂–∞–µ—Ç –æ—Ç–Ω–æ—à–µ–Ω–∏–µ –∏–Ω—Ç–µ–Ω—Å–∏–≤–Ω–æ—Å—Ç–∏ –∫–æ–Ω–≤–µ–∫—Ç–∏–≤–Ω–æ–≥–æ —Ç–µ–ø–ª–æ–æ—Ç–≤–æ–¥–∞ –∫ —Ç–µ–ø–ª–æ–ø—Ä–æ–≤–æ–¥–Ω–æ—Å—Ç–∏ –ø–ª–∞—Å—Ç–∏–Ω—ã:

$$

$$

–ó–Ω–∞—á–µ–Ω–∏–µ $Bi$ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç —Ö–∞—Ä–∞–∫—Ç–µ—Ä —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—ã –∏ –ø–æ–∑–≤–æ–ª—è–µ—Ç –æ–±–æ–±—â–∞—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –¥–ª—è —Ä–∞–∑–ª–∏—á–Ω—ã—Ö —Ñ–∏–∑–∏—á–µ—Å–∫–∏—Ö –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤.

–í–≤–µ–¥–µ–º –±–µ–∑—Ä–∞–∑–º–µ—Ä–Ω—É—é —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—É $\theta(x)$ –∏ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—É $\xi$:

$$

$$

$$

–í—ã—Ä–∞–∑–∏–º –ø—Ä–æ–∏–∑–≤–æ–¥–Ω—É—é –ø–æ $x$ —á–µ—Ä–µ–∑ –ø—Ä–æ–∏–∑–≤–æ–¥–Ω—É—é –ø–æ $\xi$:

    \frac{1}{L^2}\frac{d^2T}{d\xi^2} - \frac{hP}{kA}(T_b - T_{\infty})\theta(\xi) = 0

$$

–ò—Ç–æ–≥–æ–≤—ã–π –≤–∏–¥ –æ–±–µ–∑—Ä–∞–∑–º–µ—Ä–µ–Ω–Ω–æ–π –∑–∞–¥–∞—á–∏:
