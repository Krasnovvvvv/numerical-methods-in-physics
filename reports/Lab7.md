$$

$$

$$

$$
u_t(x,0) =
\begin{cases}

–ï—ë —Ç–æ—á–Ω–æ–µ —Ä–µ—à–µ–Ω–∏–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–∏–º–æ –≤ –≤–∏–¥–µ:

–≥–¥–µ –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç—ã —Ä—è–¥–∞ –∑–∞–¥–∞—é—Ç—Å—è –∏–Ω—Ç–µ–≥—Ä–∞–ª–æ–º –ø–æ –Ω–∞—á–∞–ª—å–Ω–æ–º—É –ø—Ä—è–º–æ—É–≥–æ–ª—å–Ω–æ–º—É –∏–º–ø—É–ª—å—Å—É —Å–∫–æ—Ä–æ—Å—Ç–∏:

$$

$$

$$
I_n = \int_a^b v_0 \sin(k_n \xi)d\xi = \frac{v_0}{k_n} \bigl(\cos(k_n a) - \cos(k_n b)\bigr),
$$

$$
a = \max(0,x_0 - \delta),\qquad

$$

#### –û–±–µ–∑—Ä–∞–∑–º–µ—Ä–∏–≤–∞–Ω–∏–µ

$$

$$

$$

$$

$$

\begin{cases}
